      \fbox{\parbox{0.9\linewidth}{#2}}

  \fi

    \ifcsname docutilsrole#1\endcsname%
      \csname docutilsrole#1\endcsname{#2}%
    \else%
      #2%
    \fi%
  \fi%
}
